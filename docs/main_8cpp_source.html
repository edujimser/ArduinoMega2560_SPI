165/**
166 * @brief ISR para cambios en el pin SS (PCINT0).
167 *
168 * - SS BAJO → inicio de paquete
169 * - SS ALTO → fin de paquete
170 */
171ISR(PCINT0_vect)
172{
173 if (SLAVE == true && SLAVE_PRUEBA == false)
174 {
175 bool ss_state = PINB & (1 << PB0);
176
177 if (!ss_state) {
178 Serial.println("SS BAJO → INICIO PAQUETE");
179 rx_index = 0;
180 receiving = true;
181 packet_ready = false;
182 } else {
183 Serial.print("SS ALTO → FIN PAQUETE. Bytes recibidos: ");
184 Serial.println(rx_index);
185 receiving = false;
186 packet_ready = true;
187 }
188 }
