36/**
37 * @enum PacketStatus
38 * @brief Códigos de estado para validación y diagnóstico de paquetes SPI.
39 */
40typedef enum {
41 PKT_VALUE_NONE = -2, /**< No se proporciona un valor asociado. */
42 PKT_PAYLOAD_NONE = -1, /**< No se especifica un índice de payload. */
43 PKT_OK = 0, /**< El paquete es válido. */
44 PKT_ERR_PAYLOAD_OUT_OF_RANGE = 1, /**< Algún payload excede el rango permitido. */
45 PKT_ERR_NULL_PTR = 2, /**< El puntero al paquete es NULL. */
46 PKT_ERR_PAYLOAD_SIZE_EXCEEDED = 3 /**< El tamaño del struct no coincide con PACKET_SIZE. */
47} PacketStatus;

49/**
50 * @struct Packet
51 * @brief Representa un paquete SPI completo.
52 *
53 * Esta estructura contiene todos los campos necesarios para transmitir
54 * un mensaje SPI según el protocolo definido.
55 */
56typedef struct {
57 uint8_t ID; /**< Identificador del comando. */
58 uint8_t LEN = PACKET_SIZE; /**< Longitud del payload (0–8). */
59 uint8_t chechSumByte_0; /**< Byte reservado para checksum (no utilizado en este protocolo). */
60 uint8_t chechSumByte_1; /**< Byte reservado para checksum (no utilizado en este protocolo). */
61 uint8_t payload_0; /**< Byte 0 del payload. */
62 uint8_t payload_1; /**< Byte 1 del payload. */
63 uint8_t payload_2; /**< Byte 2 del payload. */
64 uint8_t payload_3; /**< Byte 3 del payload. */
65 uint8_t payload_4; /**< Byte 4 del payload. */
66 uint8_t payload_5; /**< Byte 5 del payload. */
67 uint8_t payload_6; /**< Byte 6 del payload. */
68 uint8_t payload_7; /**< Byte 7 del payload. */
69 uint8_t CRC; /**< CRC-8 del paquete. */
70} Packet;

112/**
113 * @brief Calcula el checksum de 16 bits del payload del paquete.
114 *
115 * Esta función recorre los bytes del payload definidos por
116 * POSITION_PAYLOAD_BYTE y PAYLOAD_SIZE, acumulando su valor en un
117 * registro de 16 bits. El resultado final se almacena en los campos
118 * chechSumByte_0 (LSB) y chechSumByte_1 (MSB) de la estructura Packet,
119 * siguiendo el formato little-endian.
120 *
121 * @param pkt Puntero a la estructura Packet sobre la que se generará
122 * y almacenará el checksum.
123 *
124 * @pre `pkt` debe apuntar a una estructura Packet válida y con el
125 * payload correctamente inicializado.
126 *
127 * @post Los campos `chechSumByte_0` y `chechSumByte_1` contendrán el
128 * checksum de 16 bits correspondiente al payload del paquete.
129 */
130void checksum8_calculo(Packet *pkt);
