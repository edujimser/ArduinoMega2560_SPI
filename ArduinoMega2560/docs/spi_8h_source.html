1/**
2 * @file spi.h
3 * @brief API del módulo SPI para el ATmega2560 (maestro y esclavo).
4 *
5 * Este archivo declara todas las funciones públicas utilizadas para la
6 * inicialización, transferencia y diagnóstico del bus SPI. Proporciona
7 * una interfaz de alto nivel para:
8 *
9 * - Configurar el módulo SPI en modo maestro o esclavo.
10 * - Enviar y recibir bytes mediante transferencia SPI síncrona.
11 * - Enviar bloques completos de datos.
12 * - Validar estados internos del hardware SPI (SPE, SS, WCOL).
13 * - Limpiar el flag SPIF según el procedimiento del datasheet.
14 * - Mostrar información de depuración y errores SPI.
15 *
16 * Las implementaciones correspondientes se encuentran en spi.c.
17 *
18 * @note Este módulo está diseñado específicamente para el microcontrolador
19 * ATmega2560 y utiliza registros SPCR, SPSR y SPDR directamente.
20 */
21

31/**
32 * @brief Inicializa el módulo SPI en modo maestro.
33 *
34 * Configura los pines MOSI, MISO, SCK y SS, habilita SPI en modo maestro,
35 * selecciona velocidad Fosc/16, modo SPI 0 (CPOL=0, CPHA=0) y limpia SPI2X.
36 *
37 * @note Debe llamarse antes de cualquier transferencia SPI.
38 */
39void spi_init_master();
40
41/**
42 * @brief Inicializa el módulo SPI en modo esclavo.
43 *
44 * Configura los pines SPI en modo esclavo, habilita SPI y la interrupción SPIE,
45 * establece modo SPI 0 y limpia SPI2X.
46 *
47 * @note El maestro debe colocar SS en LOW para activar al esclavo.
48 */
49void spi_init_slave();
50
51/**
52 * @brief Envía un byte por SPI como maestro y recibe simultáneamente otro.
53 *
54 * Gestiona timeout, validación de SPE, estado del pin SS y colisión WCOL.
55 * Devuelve el byte recibido o un código de error SPI_ERROR_XXXX.
56 *
57 * @param data Byte a enviar.
58 * @param delay_ms Retardo opcional tras la transferencia.
59 * @return uint8_t Byte recibido o código de error.
60 */
61uint8_t spi_master_transfer(uint8_t data, uint32_t delay_ms);
62
63/**
64 * @brief Envía un bloque completo de bytes mediante SPI maestro.
65 *
66 * Realiza una transferencia por cada byte del buffer, validando errores
67 * en cada iteración (timeout, SS, SPE, WCOL). Muestra TX/RX por consola.
68 *
69 * @param data Puntero al buffer de datos a enviar.
70 * @param len Número de bytes a transmitir.
71 * @param delay_ms Retardo opcional tras cada transferencia.
72 */
73void spi_master_send_block(const uint8_t *data, size_t len, uint32_t delay_ms);
74
75/**
76 * @brief Imprime por consola la descripción del error SPI recibido.
77 *
78 * @param error_code Código de error SPI_ERROR_XXXX.
79 */
80void spi_master_print_error(uint8_t error_code);
81
82/**
83 * @brief Muestra el byte enviado y recibido durante una transferencia SPI.

85 * Incluye advertencias si el valor recibido es 0x00 o 0xFF.

87 * @param enviado Byte transmitido.
